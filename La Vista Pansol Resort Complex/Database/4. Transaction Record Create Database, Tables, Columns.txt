create database transaction_record;
use transaction_record;

CREATE TABLE `transaction_record`.`guest_transac_info` ( `ID` INT(255) NOT NULL AUTO_INCREMENT , `guestName` VARCHAR(100) NOT NULL , `guestID` VARCHAR(100) NOT NULL , `transacNo` VARCHAR(100) NOT NULL , `dateTimeIn` DATETIME NOT NULL , `dateTimeOut` DATETIME NOT NULL , `transacStatus` VARCHAR(100) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
CREATE TABLE `transaction_record`.`reservation_list` ( `ID` INT(255) NOT NULL AUTO_INCREMENT , `guestID` INT(100) NOT NULL , `transacNo` INT(100) NOT NULL , `rsvType` VARCHAR(100) NOT NULL , `rsvName` VARCHAR(100) NOT NULL , `rsvDate` DATE NOT NULL , `status` VARCHAR(100) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
CREATE TABLE `transaction_record`.`traccess` ( `ID` INT(255) NOT NULL AUTO_INCREMENT , `guestID` VARCHAR(100) NOT NULL , `transacNo` VARCHAR(100) NOT NULL , `accessType` VARCHAR(100) NOT NULL , `accessTime` VARCHAR(2) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
CREATE TABLE `transaction_record`.`trprice` ( `ID` INT(255) NOT NULL AUTO_INCREMENT , `guestID` VARCHAR(100) NOT NULL , `transacNo` VARCHAR(100) NOT NULL , `adultsPrice` INT(100) NOT NULL , `kidsPrice` INT(100) NOT NULL , `roomRate` INT(100) NOT NULL , `rentalsPrice` INT(100) NOT NULL , `accessPrice` INT(100) NOT NULL , `totalPrice` INT(100) NOT NULL , `downPayment` INT(100) NOT NULL , `totalBalance` INT(100) NOT NULL , `guestChange` INT(100) NOT NULL , `status` VARCHAR(100) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
CREATE TABLE `transaction_record`.`trrentals` ( `ID` INT(255) NOT NULL AUTO_INCREMENT , `guestID` VARCHAR(100) NOT NULL , `transacNo` VARCHAR(100) NOT NULL , `rentalName` VARCHAR(255) NOT NULL , `rentalHour` INT(2) NOT NULL , `rentalTime` VARCHAR(2) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
CREATE TABLE `transaction_record`.`trroom` ( `ID` INT(255) NOT NULL AUTO_INCREMENT , `guestID` VARCHAR(100) NOT NULL , `transacNo` VARCHAR(100) NOT NULL , `roomType` VARCHAR(100) NOT NULL , `roomName` VARCHAR(100) NOT NULL , `roomUnit` VARCHAR(100) NOT NULL , `paxRange` INT(2) NOT NULL , `kidsRange` INT(2) NOT NULL , `adultRange` INT(2) NOT NULL , `excessHour` INT(2) NOT NULL , `excessPax` INT(2) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = MyISAM;
